\section{Cài đặt Hadoop chế độ giả phân tán (Pseudo-Distributed Mode)}
\label{sec:hadoop_setup}

Trong phần này, báo cáo trình bày chi tiết từng bước cấu hình Hadoop theo chế độ \textbf{Pseudo-Distributed Mode}. Đây là chế độ triển khai mô phỏng một cụm phân tán (cluster) trên một máy tính vật lý duy nhất. Trong đó, mỗi tiến trình (daemon) của Hadoop như NameNode, DataNode, ResourceManager và NodeManager sẽ được khởi chạy độc lập dưới dạng một tiến trình Java riêng biệt, giúp người phát triển dễ dàng kiểm thử các ứng dụng Big Data trước khi triển khai thực tế.

% -------------------------------------------------------
\subsection{Bước 1: Cài đặt môi trường Java, SSH và pdsh}

\textbf{Mục đích:} Hadoop được viết bằng ngôn ngữ Java, do đó môi trường thực thi Java (Java Development Kit) là thành phần bắt buộc. Bên cạnh đó, Hadoop sử dụng giao thức SSH để điều phối các tiến trình daemon giữa các node, và công cụ \texttt{pdsh} (Parallel Distributed Shell) để chạy lệnh đồng thời trên nhiều node. Trong môi trường giả phân tán, SSH cũng được dùng để kết nối vào \texttt{localhost}.

\textbf{Các lệnh thực hiện:}
\begin{enumerate}
    \item \texttt{sudo apt update}: Cập nhật danh sách các gói phần mềm mới nhất từ kho lưu trữ của Ubuntu.
    \item \texttt{sudo apt install openjdk-11-jdk ssh pdsh -y}: Cài đặt đồng thời Java 11 JDK, OpenSSH server/client và công cụ pdsh, tham số \texttt{-y} để tự động đồng ý.
\end{enumerate}

\begin{lstlisting}[style=bashstyle, caption={Cài đặt Java 11, SSH và pdsh trên Ubuntu}]
sudo apt update
sudo apt install openjdk-11-jdk ssh pdsh -y
\end{lstlisting}

\subsection{Bước 2: Thiết lập SSH không cần mật khẩu}

\textbf{Mục đích:} Hadoop cần sử dụng chuẩn kết nối SSH để quản lý các node (thậm chí là trên cùng một máy localhost). Việc thiết lập SSH không cần mật khẩu giúp các lệnh khởi động tự động (như \texttt{start-dfs.sh}) có thể làm việc trơn tru mà không bị gián đoạn bởi yêu cầu nhập mật khẩu.

\textbf{Các lệnh thực hiện:}
\begin{enumerate}
    \item \texttt{ssh-keygen}: Tạo cặp khóa RSA (\texttt{-t rsa}), không đặt mật khẩu (\texttt{-P ''}).
    \item \texttt{cat ... >> authorized\_keys}: Đưa public key vào danh sách khóa được phép truy cập tự động.
    \item \texttt{chmod 0600}: Đặt quyền file chỉ đọc/ghi bởi chủ sở hữu (yêu cầu bảo mật SSH).
\end{enumerate}

\begin{lstlisting}[style=bashstyle, caption={Tạo SSH key và cấu hình đăng nhập tự động vào localhost}]
ssh-keygen -t rsa -P '' -f ~/.ssh/id_rsa
cat ~/.ssh/id_rsa.pub >> ~/.ssh/authorized_keys
chmod 0600 ~/.ssh/authorized_keys
\end{lstlisting}

\textbf{Kiểm tra kết nối SSH:} Thực hiện lệnh \texttt{ssh localhost} để xác nhận kết nối thành công không yêu cầu mật khẩu.

\subsection{Bước 4: Kiểm tra phiên bản và thiết lập biến môi trường}

\textbf{Mục đích:} Biến môi trường giúp hệ điều hành biết vị trí cài đặt Java và Hadoop để có thể gọi các lệnh như \texttt{hdfs}, \texttt{hadoop}, \texttt{start-dfs.sh} từ bất kỳ thư mục nào.

\textbf{Chi tiết cấu hình trong \texttt{\textasciitilde/.bashrc}:}
\begin{itemize}
    \item \texttt{JAVA\_HOME}: Trỏ tới thư mục cài đặt gốc của OpenJDK 11.
    \item \texttt{HADOOP\_HOME}: Trỏ tới thư mục vừa giải nén.
    \item \texttt{PATH}: Bổ sung các thư mục \texttt{bin} và \texttt{sbin} của Hadoop.
    \item \texttt{HADOOP\_CONF\_DIR}: Trỏ tới thư mục chứa các file cấu hình XML.
\end{itemize}

\begin{lstlisting}[style=bashstyle, caption={Cấu hình biến môi trường trong \textasciitilde/.bashrc}]
export JAVA_HOME=/usr/lib/jvm/java-11-openjdk-amd64
export HADOOP_HOME=~/hadoop
export PATH=$PATH:$HADOOP_HOME/bin:$HADOOP_HOME/sbin
export HADOOP_CONF_DIR=$HADOOP_HOME/etc/hadoop
\end{lstlisting}

\textbf{Áp dụng và kiểm tra:} Sau khi lưu file, dùng \texttt{source} để áp dụng ngay, kiểm tra bằng \texttt{java -version} và \texttt{hadoop version}.

\begin{lstlisting}[style=bashstyle, caption={Áp dụng cấu hình và kiểm tra phiên bản}]
source ~/.bashrc
java -version
hadoop version
\end{lstlisting}

\subsection{Bước 7: Xử lý sự cố \texttt{pdsh} và khởi động cụm}

\subsubsection{Lỗi kết nối \texttt{pdsh} (Permission denied)}
\textbf{Vấn đề:} Khi thực thi \texttt{start-dfs.sh} để bật các thành phần lưu trữ, lệnh bị từ chối với thông báo \texttt{Permission denied}.

\textbf{Nguyên nhân:} Mặc định Hadoop sử dụng \texttt{pdsh} ưu tiên giao thức \texttt{rsh} (Remote Shell) thay vì \texttt{ssh}. Giao thức \texttt{rsh} không hỗ trợ xác thực bằng public key RSA đã thiết lập ở Bước 2, dẫn đến từ chối kết nối.

\begin{figure}[H]
    \centering
    \includegraphics[width=0.85\textwidth]{ps_14_start_dfs_error.png}
    \caption{Lỗi \texttt{rcmd: socket: Permission denied} khi chạy \texttt{start-dfs.sh} với \texttt{pdsh} mặc định dùng \texttt{rsh}}
    \label{fig:pdsh_error}
\end{figure}

\subsubsection{Tra cứu và tìm phương án khắc phục}

Tham khảo giải pháp từ cộng đồng Stack Overflow, phương án được đánh giá cao nhất là ép buộc \texttt{pdsh} sử dụng \texttt{ssh} thay vì \texttt{rsh} thông qua biến môi trường \texttt{PDSH\_RCMD\_TYPE}.

\begin{figure}[H]
    \centering
    \includegraphics[width=\textwidth]{ps_15_pdsh_stackoverflow.png}

    \label{fig:pdsh_stackoverflow}
\end{figure}

\subsubsection{Phương án khắc phục}
\textbf{Giải quyết:} Thêm biến \texttt{PDSH\_RCMD\_TYPE=ssh} vào \texttt{\textasciitilde/.bashrc} để đảm bảo hiệu lực vĩnh viễn giữa các phiên làm việc.

\begin{lstlisting}[style=bashstyle, caption={Khắc phục lỗi pdsh bằng cách chỉ định giao thức SSH}]
echo 'export PDSH_RCMD_TYPE=ssh' >> ~/.bashrc
source ~/.bashrc
\end{lstlisting}

\subsection{Bước 9: Thao tác tệp trên HDFS và kiểm tra phân quyền}

\textbf{Mục đích:} Xây dựng hệ thống phân cấp thư mục trên HDFS, tạo người dùng và file thử nghiệm, đồng thời kiểm tra cơ chế quản lý quyền truy cập phân tán của Hadoop.

\textbf{Các lệnh và ý nghĩa:}
\begin{enumerate}
    \item \texttt{hdfs dfs -mkdir}: Tạo cây thư mục logic trên HDFS cho mã số sinh viên.
    \item \texttt{sudo useradd -m}: Tạo user Unix \texttt{khtn\_23120166} với thư mục home.
    \item \texttt{echo ... >} và \texttt{-put}: Tạo file local rồi đẩy lên HDFS bằng \texttt{-put}.
    \item \texttt{-chmod 744}: Cấp quyền đọc/ghi cho chủ sở hữu, quyền đọc cho nhóm và người khác.
    \item \texttt{-chown}: Bàn giao quyền sở hữu HDFS cho user \texttt{khtn\_23120166}.
\end{enumerate}

\begin{lstlisting}[style=bashstyle, caption={Tạo thư mục và thao tác file trên HDFS}]
hdfs dfs -mkdir /hcmus
sudo useradd -m khtn_23120166
hdfs dfs -mkdir /hcmus/23120166
echo "Hello Big Data from 23120166" > test_file.txt
hdfs dfs -put test_file.txt /hcmus/23120166/
hdfs dfs -chmod 744 /hcmus/23120166
hdfs dfs -chown -R khtn_23120166 /hcmus/23120166
hdfs dfs -ls /hcmus
\end{lstlisting}

\subsubsection{Lần chạy 3 --- Thành công, sinh file Verification}

\begin{lstlisting}[style=bashstyle, caption={Sửa quyền chủ sở hữu đệ quy và chạy jar lần 3}]
hdfs dfs -chown -R khtn_23120166 /hcmus/23120166
java -jar hadoop-test.jar 9000 /hcmus/23120166
\end{lstlisting}

\textbf{Lưu ý về cảnh báo mạng:} Trong output của lần chạy thành công, xuất hiện hai dòng cảnh báo:
\begin{quotation}
\textit{The first method to get MAC address is failed: Could not get network interface}\\[2pt]
\textit{Trying the alternative method}
\end{quotation}
Đây là hành vi bình thường trong môi trường WSL. Do kiến trúc mạng ảo hóa của WSL, máy ảo Java không thể truy xuất địa chỉ MAC trực tiếp qua giao diện mạng thông thường. Tuy nhiên, chương trình \texttt{hadoop-test.jar} đã được lập trình sẵn cơ chế dự phòng (alternative method) để thu thập thông tin mạng theo cách khác, và vẫn xuất ra file xác thực thành công.
